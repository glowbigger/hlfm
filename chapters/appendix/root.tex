\begin{proposition}
  The set of \wfs{} of \(L\) is countable.
\end{proposition}

\begin{proposition}
  An extension of \(L\) is complete and consistent if and only if, for every \(n\), \(p_n\) is a theorem of \(L\) and \((\sim p_n)\) is not, or \((\sim p_n)\) is a theorem of \(L\) and \(p_n\) is not.

  \prf{}
\end{proposition}
