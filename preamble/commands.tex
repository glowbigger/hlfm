% math number set symbols
\newcommand{\mc}[0]{\ensuremath\mathbb{C}}
\newcommand{\mr}[0]{\ensuremath\mathbb{R}}
\newcommand{\mq}[0]{\ensuremath\mathbb{Q}}
\newcommand{\mz}[0]{\ensuremath\mathbb{Z}}
\newcommand{\mn}[0]{\ensuremath\mathbb{N}}
\newcommand{\mf}[0]{\ensuremath\mathbb{F}}

% factorial with better spacing
\newcommand{\fac}[0]{\ensuremath\,!\,}

% absolute value
\newcommand{\abs}[1]{\ensuremath\lvert#1\rvert}

% wf and wfs
\newcommand{\wf}[0]{\textit{wf}}
\newcommand{\wfs}[0]{\textit{wf}s}

% ra for right arrow
% NOTE \to does the same thing, but this is more readable
\newcommand{\ra}[0]{\ensuremath\rightarrow}

\newcommand{\cub}[0]{\ensuremath\mathscr{B}}

\newcommand{\dedL}[0]{\ensuremath\vdash_{L}}
